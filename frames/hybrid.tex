\begin{frame}{混合式}{按要不要和人讨论来分工}
  按 \alert{“这一步要不要和人来回讨论”} 切分，而不是一把梭全交给 AI：
  \bigskip
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \begin{block}{人 · 交互}
        \textbf{阶段 1 · 大纲设计}

        \smallskip
        定主旨、重组逻辑、每页讲什么。最难、最需要你，反复对齐。
      \end{block}
    \column{0.48\textwidth}
      \begin{exampleblock}{AI · 确定性扇出}
        \textbf{阶段 2 找图 / 阶段 3 排版}

        \smallskip
        目标定死后并发干，每图、每页各一个 agent。
      \end{exampleblock}
  \end{columns}
  \bigskip
  \narration{旁白：子 agent 中途无法问你，所以把要讨论的都收在阶段 1。}
\end{frame}
